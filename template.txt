    <table>
        <thead>
            <tr>
                <th>NIC</th>
                <th>Bytes Sent</th>
                <th>Bytes Received</th>
                <th>Packets Sent</th>
                <th>Packets Received</th>
                <th>Errors In</th>
                <th>Errors Out</th>
                <th>Drop In</th>
                <th>Drop Out</th>
            </tr>
        </thead>
        <tbody id="network-stats"></tbody>
    </table>
    <script>
        async function fetchNetworkStats() {
            const response = await fetch('/api/network_stats');
            const data = await response.json();
            const tableBody = document.getElementById('network-stats');
            tableBody.innerHTML = '';
            for (const [nic, stats] of Object.entries(data)) {
                const row = document.createElement('tr');
                row.innerHTML = `
                    <td>${nic}</td>
                    <td>${stats.bytes_sent}</td>
                    <td>${stats.bytes_recv}</td>
                    <td>${stats.packets_sent}</td>
                    <td>${stats.packets_recv}</td>
                    <td>${stats.errin}</td>
                    <td>${stats.errout}</td>
                    <td>${stats.dropin}</td>
                    <td>${stats.dropout}</td>
                `;
                tableBody.appendChild(row);
            }
        }

        setInterval(fetchNetworkStats, 5000);
        fetchNetworkStats();
    </script>
